using FHIR version '4.0.1'

include ANCConfig called Config
include ANCConcepts called Cx
include ANCDataElements called PatientData
include ANCContactDataElements called ContactData

context Patient

/*
(Minimum requirements for IPV assessment = TRUE
)
  AND (Woman discloses or is suspected to be subjected to intimate partner violence = TRUE)
*/
define &quot;Should Conduct clinical enquiry&quot;:
  Config.&quot;Minimum requirements for IPV assessment&quot; = true
    and ContactData.&quot;Woman discloses or is suspected to be subjected to intimate partner violence&quot; is true

/*
(Minimum requirements for IPV assessment = TRUE)
  AND (&quot;Presenting signs and symptoms that trigger suspicion of IPV and clinical inquiry&quot; = &quot;Ongoing stress&quot;)
*/
define &quot;Should Conduct clinical enquiry and provide first-line support and care if IPV confirmed&quot;:
  Config.&quot;Minimum requirements for IPV assessment&quot; = true
    and ContactData.&quot;Presenting signs and symptoms that trigger suspicion of IPV and clinical inquiry&quot; in Cx.&quot;Presenting signs and symptoms that trigger suspicion of IPV and clinical inquiry - Ongoing stress Choices&quot;
